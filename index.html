Progam w C

Program wykonuje rekurencyjną "n" silnię

	
	
    		#include< stdio.h >
    		#include< stdlib.h >

    		int silnia(int a)
    		{
      			if ( a == 0 ) return 1;
      			else return a * silnia(a-1);
    		}

    		int main ()
    		{
	    		printf("Silnia - %d", silnia(3));
	    
	    		getchar();
	
	    		return 0;
    		}
